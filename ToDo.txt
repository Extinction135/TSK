Gif Capture process:
	run game in release mode
	run LiceCap, expand capture area past program window about 10 pixels on each side
	record at 30fps, open in photoshop and crop the frame down to just the game view (minus window title)
	reduce image size by half, down to original size so pixels are 1:1 ratio
	save as PSD master, then begin cutting out sections/edits
	the default time per frame (set by licecap) is perfect, do not change this
	save for web ->  GIF 32 Dithered, filesize should be under 3mb, loop forever
	test in browser to ensure proper fps

SetDitch(metaDitch);

-neighbors is top to bottom, 8 neighboring directions.
SetDitch(GameObject Ditch, List<Boolean> Neighbors)
{

	if neigh.allFalse, Ditch = single hole

	if neigh.N & neigh.S & othersFalse, Ditch = vertical straight
	if neigh.E & neigh.W & othersFalse, Ditch = horizontal straight

	if neigh.E & neigh.N & othersFalse, Ditch = corner
	if neigh.E & neigh.S & othersFalse, Ditch = corner
	if neigh.W & neigh.N & othersFalse, Ditch = corner
	if neigh.W & neigh.S & othersFalse, Ditch = corner

	if neigh.W & neigh.N & neigh.E & othersFalse, Ditch = T connector
	if neigh.W & neigh.S & neigh.E & othersFalse, Ditch = T connector
	if neigh.W & neigh.S & neigh.N & othersFalse, Ditch = T connector
	if neigh.E & neigh.S & neigh.N & othersFalse, Ditch = T connector

	if neigh.N & neigh.S & neigh.E & neigh.W, Ditch = 4 connector

}
